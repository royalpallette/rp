import { storage, db } from '../js/firebase-config.js'; import { ref as storageRef, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-storage.js"; import { ref as dbRef, set, onValue } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-database.js"; const form = document.getElementById('favicon-form'); const saveBtn = document.getElementById('save-btn'); const statusMsg = document.getElementById('status-msg'); // Load current favicon onValue(dbRef(db, 'settings/favicon'), (snapshot) => { if (snapshot.exists()) { const url = snapshot.val(); document.getElementById('favicon-preview').src = url; document.getElementById('favicon-url-text').textContent = "Active"; } else { document.getElementById('favicon-url-text').textContent = "No custom favicon set."; } }); form.addEventListener('submit', async (e) => { e.preventDefault(); const file = document.getElementById('favicon-file').files[0]; if (!file) return; saveBtn.disabled = true; saveBtn.textContent = 'Uploading...'; statusMsg.classList.add('hidden'); try { // Upload to Storage const ext = file.name.split('.').pop(); const sRef = storageRef(storage, `site_settings/favicon_${Date.now()}.${ext}`); await uploadBytes(sRef, file); // Get URL const url = await getDownloadURL(sRef); // Save to RTDB await set(dbRef(db, 'settings/favicon'), url); statusMsg.textContent = "✅ Favicon updated successfully! Refresh your website to see the change."; statusMsg.className = "text-sm text-green-600 block mt-2"; form.reset(); } catch (error) { console.error("Upload error", error); statusMsg.textContent = "❌ Error uploading: " + error.message; statusMsg.className = "text-sm text-red-600 block mt-2"; } finally { saveBtn.disabled = false; saveBtn.textContent = 'Upload & Save Favicon'; } }); // --- Receipt Settings Logic --- const receiptForm = document.getElementById('receipt-form'); const recSaveBtn = document.getElementById('save-receipt-btn'); const recStatus = document.getElementById('receipt-status-msg'); onValue(dbRef(db, 'settings/receipt'), (snapshot) => { if (snapshot.exists()) { const data = snapshot.val(); document.getElementById('rec-shop-name').value = data.shopName || ''; document.getElementById('rec-shop-address').value = data.shopAddress || ''; document.getElementById('rec-shop-phone').value = data.shopPhone || ''; document.getElementById('rec-footer-msg').value = data.footerMsg || ''; } }); receiptForm.addEventListener('submit', async (e) => { e.preventDefault(); recSaveBtn.disabled = true; recSaveBtn.textContent = 'Saving...'; recStatus.classList.add('hidden'); const data = { shopName: document.getElementById('rec-shop-name').value, shopAddress: document.getElementById('rec-shop-address').value, shopPhone: document.getElementById('rec-shop-phone').value, footerMsg: document.getElementById('rec-footer-msg').value }; try { await set(dbRef(db, 'settings/receipt'), data); recStatus.textContent = "✅ Receipt settings saved successfully!"; recStatus.className = "text-sm text-green-600 block mt-2"; } catch(error) { console.error(error); recStatus.textContent = "❌ Error saving: " + error.message; recStatus.className = "text-sm text-red-600 block mt-2"; } finally { recSaveBtn.disabled = false; recSaveBtn.textContent = 'Save Receipt Settings'; } });
